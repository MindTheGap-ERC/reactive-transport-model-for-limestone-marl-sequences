149    def calculate_sigma(Peclet, W_data, Peclet_min, Peclet_max):
150        # Assuming the arrays are 1D.
151        sigma = np.empty(Peclet.size)
152        for i in range(sigma.size):
153            if np.abs(Peclet[i]) < Peclet_min:
154                sigma[i] = 0
155            elif np.abs(Peclet[i]) > Peclet_max:
156                sigma[i] = np.sign(W_data[i])
157            else:
158                sigma[i] = np.cosh(Peclet[i])/np.sinh(Peclet[i]) - \
159                    1/Peclet[i]
160        return sigma

162    def fun(self, t, y, progress_proxy, progress_dt, t0):
163        """ 
164        For tqdm to monitor progress. 
165        From 
166        https://stackoverflow.com/questions/59047892/
167        how-to-monitor-the-process-of-scipy-odeint """
168        if self.last_t == 0.:
169            self.last_t = t0        
170        n = int((t - self.last_t) / progress_dt)
171        progress_proxy.update(n)
172        self.last_t += n * progress_dt        
173        
174        CA = ScalarField(self.Depths, y[self.slices_all_fields[0]])
175        CC = ScalarField(self.Depths, y[self.slices_all_fields[1]])
176        cCa = ScalarField(self.Depths, y[self.slices_all_fields[2]])
177        cCO3 = ScalarField(self.Depths, y[self.slices_all_fields[3]])
178        Phi = ScalarField(self.Depths, y[self.slices_all_fields[4]])
179
180        two_factors = cCa * cCO3
181        two_factors_upp_lim = two_factors.to_scalar(lambda f: np.fmin(f,1))
182        two_factors_low_lim = two_factors.to_scalar(lambda f: np.fmax(f,1))
183
184        three_factors = two_factors * self.KRat
185        three_factors_upp_lim = three_factors.to_scalar(lambda f: np.fmin(f,1))
186        three_factors_low_lim = three_factors.to_scalar(lambda f: np.fmax(f,1))
187
188        coA = CA * (((1 - three_factors_upp_lim) ** self.m2) * \
189                    (self.not_too_deep * self.not_too_shallow) - self.nu1 * \
190                    (three_factors_low_lim - 1) ** self.m1)
191 
192        coC = CC * (((two_factors_low_lim - 1) ** self.n1) - self.nu2 * \
193                    (1 - two_factors_upp_lim) ** self.n2)
194
195        F = 1 - np.exp(10 - 10 / Phi)
196
197        U = self.presum + self.rhorat * Phi ** 3 * F / (1 - Phi)
198        
199        # Instead of the default central differenced gradient from py-pde
200        # construct forward and backward differenced gradients and apply
201        # either one of them, based on the sign of U.
202        CA_grad_back = CA.apply_operator("grad_back", self.bc_CA)
203        CA_grad_forw = CA.apply_operator("grad_forw", self.bc_CA)
204        CA_grad = ScalarField(self.Depths, np.where(U.data>0, CA_grad_back.data, \
205            CA_grad_forw.data))
206
207        CC_grad_back = CC.apply_operator("grad_back", self.bc_CC)
208        CC_grad_forw = CC.apply_operator("grad_forw", self.bc_CC)    
209        CC_grad = ScalarField(self.Depths, np.where(U.data>0, CC_grad_back.data, \
210            CC_grad_forw.data))
211
212        W = self.presum - self.rhorat * Phi ** 2 * F
213        
214        dCA_dt = - U * CA_grad - self.Da * ((1 - CA) \
215                 * coA + self.lambda_ * CA * coC)
216
217        dCC_dt = - U * CC_grad + self.Da * (self.lambda_ \
218                 * (1 - CC) * coC + CC * coA)
219
220        # Implementing equation 6 from l'Heureux.
221        denominator = 1 - 2 * np.log(Phi)
222        # dPhi = self.auxcon * F * (Phi ** 3) / (1 - Phi)
223        dPhi = self.dPhi_fixed
224
225        if self.FV_switch:
226            # Fiadeiro-Veronis scheme for equations 42 and 43
227            # from l'Heureux. 
228            common_Peclet  = W * self.delta_x / 2. 
229            Peclet_cCa =  common_Peclet * denominator/ self.dCa       
230            sigma_cCa_data = LMAHeureuxPorosityDiff.calculate_sigma(\
231                Peclet_cCa.data, W.data, self.Peclet_min, self.Peclet_max)
232            sigma_cCa = ScalarField(self.Depths, sigma_cCa_data)
233
234            Peclet_cCO3 = common_Peclet * denominator/ self.dCO3      
235            sigma_cCO3_data = LMAHeureuxPorosityDiff.calculate_sigma(\
236                Peclet_cCO3.data, W.data, self.Peclet_min, self.Peclet_max)
237            sigma_cCO3 = ScalarField(self.Depths, sigma_cCO3_data)
238
239            Peclet_Phi = common_Peclet / dPhi
240            sigma_Phi_data = LMAHeureuxPorosityDiff.calculate_sigma(\
241                Peclet_Phi.data, W.data, self.Peclet_min, self.Peclet_max)
242            sigma_Phi = ScalarField(self.Depths, sigma_Phi_data)
243        else:
244            sigma_cCa = 0
245            sigma_cCO3 = 0
246            sigma_Phi = 0
247
248        # Instead of the default central differenced gradient from py-pde
249        # construct forward and backward differenced gradients and give them
250        # appropriate weights according to a Fiadeiro-Veronis scheme.
251        cCa_grad_back = cCa.apply_operator("grad_back", self.bc_cCa)
252        cCa_grad_forw = cCa.apply_operator("grad_forw", self.bc_cCa)      
253        cCa_grad = 0.5 * ((1-sigma_cCa) * cCa_grad_forw +\
254             (1+sigma_cCa) * cCa_grad_back)
255        cCa_laplace = cCa.laplace(self.bc_cCa)
256
257        cCO3_grad_back = cCO3.apply_operator("grad_back", self.bc_cCO3)
258        cCO3_grad_forw = cCO3.apply_operator("grad_forw", self.bc_cCO3)
259        cCO3_grad = 0.5 * ((1-sigma_cCO3) * cCO3_grad_forw +\
260             (1+sigma_cCO3) * cCO3_grad_back)
261        cCO3_laplace = cCO3.laplace(self.bc_cCO3)
262
263        Phi_grad_back = Phi.apply_operator("grad_back", self.bc_Phi)
264        Phi_grad_forw = Phi.apply_operator("grad_forw", self.bc_Phi)
265        Phi_grad = 0.5 * ((1-sigma_Phi) * Phi_grad_forw +\
266             (1+sigma_Phi) * Phi_grad_back)
267        Phi_laplace = Phi.laplace(self.bc_Phi)
268
269        Phi_denom = Phi/denominator
270        grad_Phi_denom = Phi_grad * (denominator + 2) / denominator ** 2
271
272        common_helper = coA - self.lambda_ * coC
273       
274        dcCa_dt = (cCa_grad * grad_Phi_denom + Phi_denom * cCa_laplace) \
275                  * self.dCa /Phi - W * cCa_grad \
276                  + self.Da * (1 - Phi) * (self.delta - cCa) * common_helper / Phi
277
278        dcCO3_dt = (cCO3_grad * grad_Phi_denom + Phi_denom * cCO3_laplace) \
279                   * self.dCO3/Phi - W * cCO3_grad \
280                   + self.Da * (1 - Phi) * (self.delta - cCO3) * common_helper / Phi
281
282        dW_dx = -self.rhorat * Phi_grad * (2 * Phi * F + 10 * (F - 1))
283
284        dPhi_dt = - (Phi * dW_dx + W * Phi_grad) \
285                  + dPhi * Phi_laplace \
286                  + self.Da * (1 - Phi) * common_helper
287
288        return FieldCollection([dCA_dt, dCC_dt, dcCa_dt, dcCO3_dt, dPhi_dt]).data.ravel()

410        cCO3_grad = np.empty(no_depths)
411        Phi_grad = np.empty(no_depths)
412
413        for i in range(no_depths):
414            F[i] = 1 - np.exp(10 - 10 / Phi[i])
415
416            U[i] = presum + rhorat * Phi[i] ** 3 * F[i]/ (1 - Phi[i])
417
418            if U[i] > 0:
419                CA_grad[i] = CA_grad_back[i]
420                CC_grad[i] = CC_grad_back[i]
421            else:
422                CA_grad[i] = CA_grad_forw[i]
423                CC_grad[i] = CC_grad_forw[i]
424
425            W[i] = presum - rhorat * Phi[i] ** 2 * F[i]
426
427            # Implementing equation 6 from l'Heureux.
428            denominator[i] = 1 - 2 * np.log(Phi[i])
429            one_minus_Phi[i] = 1 - Phi[i]
430            # dPhi[i] = auxcon * F[i] * (Phi[i] ** 3) / one_minus_Phi[i]
431            dPhi[i] = dPhi_fixed
432
433            if FV_switch:
434                # Fiadeiro-Veronis scheme for equations 42 and 43
435                # from l'Heureux. 
436                Peclet_cCa = W[i] * delta_x * denominator[i]/ (2. * dCa )
437                if np.abs(Peclet_cCa) < Peclet_min:
438                    sigma_cCa = 0
439                elif np.abs(Peclet_cCa) > Peclet_max:
440                    sigma_cCa = np.sign(W[i])
441                else:
442                     sigma_cCa = np.cosh(Peclet_cCa)/np.sinh(Peclet_cCa) - 1/Peclet_cCa
443
444                Peclet_cCO3 = W[i] * delta_x * denominator[i]/ (2. * dCO3)
445                if np.abs(Peclet_cCO3) < Peclet_min:
446                    sigma_cCO3 = 0
447                elif np.abs(Peclet_cCO3) > Peclet_max:
448                    sigma_cCO3 = np.sign(W[i])
449                else:
450                    sigma_cCO3 = np.cosh(Peclet_cCO3)/np.sinh(Peclet_cCO3) - 1/Peclet_cCO3
451
452                Peclet_Phi = W[i] * delta_x / (2. * dPhi[i])
453                if np.abs(Peclet_Phi) < Peclet_min:
454                    sigma_Phi = 0
455                elif np.abs(Peclet_Phi) > Peclet_max:
456                    sigma_Phi = np.sign(W[i])
457                else:
458                    sigma_Phi = np.cosh(Peclet_Phi)/np.sinh(Peclet_Phi) - 1/Peclet_Phi
459            else:
460                sigma_cCa = 0
461                sigma_cCO3 = 0
462                sigma_Phi = 0
463
464            cCa_grad[i] = 0.5 * ((1-sigma_cCa) * cCa_grad_forw[i] + \
465                          (1+sigma_cCa) * cCa_grad_back[i])
466            cCO3_grad[i] = 0.5 * ((1-sigma_cCO3) * cCO3_grad_forw[i] + \
467                          (1+sigma_cCO3) * cCO3_grad_back[i])
468            Phi_grad[i] = 0.5 * ((1-sigma_Phi) * Phi_grad_forw [i]+ \
469                          (1+sigma_Phi) * Phi_grad_back[i])
470
471            common_helper1[i] = Phi[i]/denominator[i]
472            common_helper2[i] = Phi_grad[i] * (2 + denominator[i]) \
473                                / denominator[i] ** 2
474            helper_cCa_grad[i] = dCa * (common_helper2[i] * cCa_grad[i] \
475                                 + common_helper1[i] * cCa_laplace[i])
476            helper_cCO3_grad[i] = dCO3 * (common_helper2[i] * cCO3_grad[i] \
477                                 + common_helper1[i] * cCO3_laplace[i])            
478
479            two_factors[i] = cCa[i] * cCO3[i]
480            two_factors_upp_lim[i] = min(two_factors[i],1)
481            two_factors_low_lim[i] = max(two_factors[i],1)
482            three_factors[i] = two_factors[i] * KRat
483            three_factors_upp_lim[i] = min(three_factors[i],1)
484            three_factors_low_lim[i] = max(three_factors[i],1)
485
486            coA[i] = CA[i] * (((1 - three_factors_upp_lim[i]) ** m2) * \
487                (not_too_deep[i] * not_too_shallow[i]) - nu1 * \
488                (three_factors_low_lim[i] - 1) ** m1)
489
490            coC[i] = CC[i] * (((two_factors_low_lim[i] - 1) ** n1) - nu2 * \
491                (1 - two_factors_upp_lim[i]) ** n2)         
492
493            common_helper3[i] = coA[i] - lambda_* coC[i]
494               
495            dW_dx[i] = -rhorat * Phi_grad[i] * (2 * Phi[i] * F[i] + 10 * (F[i] - 1))    
496       
497            # This is dCA_dt
498            rate[i] = - U[i] * CA_grad[i] - Da * ((1 - CA[i]) \
499                                    * coA[i] + lambda_ * CA[i] * coC[i])
500
501            # This is dCC_dt
502            rate[no_depths + i] = - U[i] * CC_grad[i] + Da * (lambda_ * \
503                                    (1 - CC[i]) * coC[i] + CC[i] * coA[i])              
504
505            # This is dcCa_dt
506            rate[2 * no_depths + i] =  helper_cCa_grad[i]/Phi[i] - W[i] * \
507                                        cCa_grad[i] + Da * one_minus_Phi[i] * \
508                                        (delta - cCa[i]) * common_helper3[i] \
509                                        /Phi[i]                                 
510
511            # This is dcCO3_dt
512            rate[3 * no_depths + i] =  helper_cCO3_grad[i]/Phi[i] - W[i] * \
513                                        cCO3_grad[i] + Da * one_minus_Phi[i] * \
514                                        (delta - cCO3[i]) * common_helper3[i] \
515                                        /Phi[i]                       
516
517            # This is dPhi_dt
518            rate[4 * no_depths + i] = - (dW_dx[i] * Phi[i] + W[i] * Phi_grad[i]) \
519                                      + dPhi[i] * Phi_laplace[i] + Da * one_minus_Phi[i] \
520                                      * common_helper3[i] 
521
522        return rate

163    def fun(self, t, y, progress_proxy, progress_dt, t0):
164        """ 
165        For tqdm to monitor progress. 
166        From 
167        https://stackoverflow.com/questions/59047892/
168        how-to-monitor-the-process-of-scipy-odeint """
169        if self.last_t == 0.:
170            self.last_t = t0        
171        n = int((t - self.last_t) / progress_dt)
172        progress_proxy.update(n)
173        self.last_t += n * progress_dt        
174        
175        CA = ScalarField(self.Depths, y[self.slices_all_fields[0]])
176        CC = ScalarField(self.Depths, y[self.slices_all_fields[1]])
177        cCa = ScalarField(self.Depths, y[self.slices_all_fields[2]])
178        cCO3 = ScalarField(self.Depths, y[self.slices_all_fields[3]])
179        Phi = ScalarField(self.Depths, y[self.slices_all_fields[4]])
180
181        two_factors = cCa * cCO3
182        two_factors_upp_lim = two_factors.to_scalar(lambda f: np.fmin(f,1))
183        two_factors_low_lim = two_factors.to_scalar(lambda f: np.fmax(f,1))
184
185        three_factors = two_factors * self.KRat
186        three_factors_upp_lim = three_factors.to_scalar(lambda f: np.fmin(f,1))
187        three_factors_low_lim = three_factors.to_scalar(lambda f: np.fmax(f,1))
188
189        coA = CA * (((1 - three_factors_upp_lim) ** self.m2) * \
190                    (self.not_too_deep * self.not_too_shallow) - self.nu1 * \
191                    (three_factors_low_lim - 1) ** self.m1)
192 
193        coC = CC * (((two_factors_low_lim - 1) ** self.n1) - self.nu2 * \
194                    (1 - two_factors_upp_lim) ** self.n2)
195
196        F = 1 - np.exp(10 - 10 / Phi)
197
198        U = self.presum + self.rhorat * Phi ** 3 * F / (1 - Phi)
199        
200        # Instead of the default central differenced gradient from py-pde
201        # construct forward and backward differenced gradients and apply
202        # either one of them, based on the sign of U.
203        CA_grad_back = CA.apply_operator("grad_back", self.bc_CA)
204        CA_grad_forw = CA.apply_operator("grad_forw", self.bc_CA)
205        CA_grad = ScalarField(self.Depths, np.where(U.data>0, CA_grad_back.data, \
206            CA_grad_forw.data))
207
208        CC_grad_back = CC.apply_operator("grad_back", self.bc_CC)
209        CC_grad_forw = CC.apply_operator("grad_forw", self.bc_CC)    
210        CC_grad = ScalarField(self.Depths, np.where(U.data>0, CC_grad_back.data, \
211            CC_grad_forw.data))
212
213        W = self.presum - self.rhorat * Phi ** 2 * F
214        
215        dCA_dt = - U * CA_grad - self.Da * ((1 - CA) \
216                 * coA + self.lambda_ * CA * coC)
217
218        dCC_dt = - U * CC_grad + self.Da * (self.lambda_ \
219                 * (1 - CC) * coC + CC * coA)
220
221        # Implementing equation 6 from l'Heureux.
222        denominator = 1 - 2 * np.log(Phi)
223        # dPhi = self.auxcon * F * (Phi ** 3) / (1 - Phi)
224        dPhi = self.dPhi_fixed
225
226        if self.FV_switch:
227            # Fiadeiro-Veronis scheme for equations 42 and 43
228            # from l'Heureux. 
229            common_Peclet  = W * self.delta_x / 2. 
230            Peclet_cCa =  common_Peclet * denominator/ self.dCa       
231            sigma_cCa_data = LMAHeureuxPorosityDiff.calculate_sigma(\
232                Peclet_cCa.data, W.data, self.Peclet_min, self.Peclet_max)
233            sigma_cCa = ScalarField(self.Depths, sigma_cCa_data)
234
235            Peclet_cCO3 = common_Peclet * denominator/ self.dCO3      
236            sigma_cCO3_data = LMAHeureuxPorosityDiff.calculate_sigma(\
237                Peclet_cCO3.data, W.data, self.Peclet_min, self.Peclet_max)
238            sigma_cCO3 = ScalarField(self.Depths, sigma_cCO3_data)
239
240            Peclet_Phi = common_Peclet / dPhi
241            sigma_Phi_data = LMAHeureuxPorosityDiff.calculate_sigma(\
242                Peclet_Phi.data, W.data, self.Peclet_min, self.Peclet_max)
243            sigma_Phi = ScalarField(self.Depths, sigma_Phi_data)
244        else:
245            sigma_cCa = 0
246            sigma_cCO3 = 0
247            sigma_Phi = 0
248
249        # Instead of the default central differenced gradient from py-pde
250        # construct forward and backward differenced gradients and give them
251        # appropriate weights according to a Fiadeiro-Veronis scheme.
252        cCa_grad_back = cCa.apply_operator("grad_back", self.bc_cCa)
253        cCa_grad_forw = cCa.apply_operator("grad_forw", self.bc_cCa)      
254        cCa_grad = 0.5 * ((1-sigma_cCa) * cCa_grad_forw +\
255             (1+sigma_cCa) * cCa_grad_back)
256        cCa_laplace = cCa.laplace(self.bc_cCa)
257
258        cCO3_grad_back = cCO3.apply_operator("grad_back", self.bc_cCO3)
259        cCO3_grad_forw = cCO3.apply_operator("grad_forw", self.bc_cCO3)
260        cCO3_grad = 0.5 * ((1-sigma_cCO3) * cCO3_grad_forw +\
261             (1+sigma_cCO3) * cCO3_grad_back)
262        cCO3_laplace = cCO3.laplace(self.bc_cCO3)
263
264        Phi_grad_back = Phi.apply_operator("grad_back", self.bc_Phi)
265        Phi_grad_forw = Phi.apply_operator("grad_forw", self.bc_Phi)
266        Phi_grad = 0.5 * ((1-sigma_Phi) * Phi_grad_forw +\
267             (1+sigma_Phi) * Phi_grad_back)
268        Phi_laplace = Phi.laplace(self.bc_Phi)
269
270        Phi_denom = Phi/denominator
271        grad_Phi_denom = Phi_grad * (denominator + 2) / denominator ** 2
272
273        common_helper = coA - self.lambda_ * coC
274       
275        dcCa_dt = (cCa_grad * grad_Phi_denom + Phi_denom * cCa_laplace) \
276                  * self.dCa /Phi - W * cCa_grad \
277                  + self.Da * (1 - Phi) * (self.delta - cCa) * common_helper / Phi
278
279        dcCO3_dt = (cCO3_grad * grad_Phi_denom + Phi_denom * cCO3_laplace) \
280                   * self.dCO3/Phi - W * cCO3_grad \
281                   + self.Da * (1 - Phi) * (self.delta - cCO3) * common_helper / Phi
282
283        dW_dx = -self.rhorat * Phi_grad * (2 * Phi * F + 10 * (F - 1))
284
285        dPhi_dt = - (Phi * dW_dx + W * Phi_grad) \
286                  + dPhi * Phi_laplace \
287                  + self.Da * (1 - Phi) * common_helper
288
289        return FieldCollection([dCA_dt, dCC_dt, dcCa_dt, dcCO3_dt, dPhi_dt]).data.ravel()

364    def pde_rhs(CA, CC, cCa, cCO3, Phi, CA_grad_back_op, CA_grad_forw_op,
365                CC_grad_back_op, CC_grad_forw_op,
366                cCa_grad_back_op, cCa_grad_forw_op, cCa_laplace_op, 
367                cCO3_grad_back_op, cCO3_grad_forw_op, cCO3_laplace_op, 
368                Phi_grad_back_op, Phi_grad_forw_op, Phi_laplace_op, 
369                KRat, m1, m2, n1, n2, nu1, nu2, not_too_deep, not_too_shallow, 
370                presum, rhorat, lambda_, Da, dCa, dCO3, delta, auxcon, delta_x, 
371                Peclet_min, Peclet_max, no_depths, dPhi_fixed, FV_switch):
372
373        CA_grad_back = CA_grad_back_op(CA) 
374        CA_grad_forw = CA_grad_forw_op(CA) 
375        CC_grad_back = CC_grad_back_op(CC)
376        CC_grad_forw = CC_grad_forw_op(CC)
377        cCa_grad_back = cCa_grad_back_op(cCa)
378        cCa_grad_forw = cCa_grad_forw_op(cCa)
379        cCa_laplace = cCa_laplace_op(cCa)
380        cCO3_grad_back = cCO3_grad_back_op(cCO3)
381        cCO3_grad_forw = cCO3_grad_forw_op(cCO3)
382        cCO3_laplace = cCO3_laplace_op(cCO3)
383        Phi_grad_back = Phi_grad_back_op(Phi) 
384        Phi_grad_forw = Phi_grad_forw_op(Phi)
385        Phi_laplace = Phi_laplace_op(Phi)
386
387        denominator = np.empty(no_depths)
388        common_helper1 = np.empty(no_depths)
389        common_helper2 = np.empty(no_depths)
390        helper_cCa_grad = np.empty(no_depths)
391        helper_cCO3_grad = np.empty(no_depths)
392        F = np.empty(no_depths)
393        U = np.empty(no_depths)
394        W = np.empty(no_depths)
395        two_factors = np.empty(no_depths)
396        two_factors_upp_lim = np.empty(no_depths)
397        two_factors_low_lim = np.empty(no_depths)
398        three_factors = np.empty(no_depths)
399        three_factors_upp_lim = np.empty(no_depths)
400        three_factors_low_lim = np.empty(no_depths)
401        coA = np.empty(no_depths)
402        coC = np.empty(no_depths)
403        common_helper3 = np.empty(no_depths)
404        rate = np.empty(5 * no_depths)
405        dPhi = np.empty(no_depths)
406        dW_dx = np.empty(no_depths)
407        one_minus_Phi = np.empty(no_depths)
408        CA_grad = np.empty(no_depths)
409        CC_grad = np.empty(no_depths)
410        cCa_grad = np.empty(no_depths)
411        cCO3_grad = np.empty(no_depths)
412        Phi_grad = np.empty(no_depths)
413
414        for i in range(no_depths):
415            F[i] = 1 - np.exp(10 - 10 / Phi[i])
416
417            U[i] = presum + rhorat * Phi[i] ** 3 * F[i]/ (1 - Phi[i])
418
419            if U[i] > 0:
420                CA_grad[i] = CA_grad_back[i]
421                CC_grad[i] = CC_grad_back[i]
422            else:
423                CA_grad[i] = CA_grad_forw[i]
424                CC_grad[i] = CC_grad_forw[i]
425
426            W[i] = presum - rhorat * Phi[i] ** 2 * F[i]
427
428            # Implementing equation 6 from l'Heureux.
429            denominator[i] = 1 - 2 * np.log(Phi[i])
430            one_minus_Phi[i] = 1 - Phi[i]
431            # dPhi[i] = auxcon * F[i] * (Phi[i] ** 3) / one_minus_Phi[i]
432            dPhi[i] = dPhi_fixed
433
434            if FV_switch:
435                # Fiadeiro-Veronis scheme for equations 42 and 43
436                # from l'Heureux. 
437                Peclet_cCa = W[i] * delta_x * denominator[i]/ (2. * dCa )
438                if np.abs(Peclet_cCa) < Peclet_min:
439                    sigma_cCa = 0
440                elif np.abs(Peclet_cCa) > Peclet_max:
441                    sigma_cCa = np.sign(W[i])
442                else:
443                     sigma_cCa = np.cosh(Peclet_cCa)/np.sinh(Peclet_cCa) - 1/Peclet_cCa
444
445                Peclet_cCO3 = W[i] * delta_x * denominator[i]/ (2. * dCO3)
446                if np.abs(Peclet_cCO3) < Peclet_min:
447                    sigma_cCO3 = 0
448                elif np.abs(Peclet_cCO3) > Peclet_max:
449                    sigma_cCO3 = np.sign(W[i])
450                else:
451                    sigma_cCO3 = np.cosh(Peclet_cCO3)/np.sinh(Peclet_cCO3) - 1/Peclet_cCO3
452
453                Peclet_Phi = W[i] * delta_x / (2. * dPhi[i])
454                if np.abs(Peclet_Phi) < Peclet_min:
455                    sigma_Phi = 0
456                elif np.abs(Peclet_Phi) > Peclet_max:
457                    sigma_Phi = np.sign(W[i])
458                else:
459                    sigma_Phi = np.cosh(Peclet_Phi)/np.sinh(Peclet_Phi) - 1/Peclet_Phi
460            else:
461                sigma_cCa = 0
462                sigma_cCO3 = 0
463                sigma_Phi = 0
464
465            cCa_grad[i] = 0.5 * ((1-sigma_cCa) * cCa_grad_forw[i] + \
466                          (1+sigma_cCa) * cCa_grad_back[i])
467            cCO3_grad[i] = 0.5 * ((1-sigma_cCO3) * cCO3_grad_forw[i] + \
468                          (1+sigma_cCO3) * cCO3_grad_back[i])
469            Phi_grad[i] = 0.5 * ((1-sigma_Phi) * Phi_grad_forw [i]+ \
470                          (1+sigma_Phi) * Phi_grad_back[i])
471
472            common_helper1[i] = Phi[i]/denominator[i]
473            common_helper2[i] = Phi_grad[i] * (2 + denominator[i]) \
474                                / denominator[i] ** 2
475            helper_cCa_grad[i] = dCa * (common_helper2[i] * cCa_grad[i] \
476                                 + common_helper1[i] * cCa_laplace[i])
477            helper_cCO3_grad[i] = dCO3 * (common_helper2[i] * cCO3_grad[i] \
478                                 + common_helper1[i] * cCO3_laplace[i])            
479
480            two_factors[i] = cCa[i] * cCO3[i]
481            two_factors_upp_lim[i] = min(two_factors[i],1)
482            two_factors_low_lim[i] = max(two_factors[i],1)
483            three_factors[i] = two_factors[i] * KRat
484            three_factors_upp_lim[i] = min(three_factors[i],1)
485            three_factors_low_lim[i] = max(three_factors[i],1)
486
487            coA[i] = CA[i] * (((1 - three_factors_upp_lim[i]) ** m2) * \
488                (not_too_deep[i] * not_too_shallow[i]) - nu1 * \
489                (three_factors_low_lim[i] - 1) ** m1)
490
491            coC[i] = CC[i] * (((two_factors_low_lim[i] - 1) ** n1) - nu2 * \
492                (1 - two_factors_upp_lim[i]) ** n2)         
493
494            common_helper3[i] = coA[i] - lambda_* coC[i]
495               
496            dW_dx[i] = -rhorat * Phi_grad[i] * (2 * Phi[i] * F[i] + 10 * (F[i] - 1))    
497       
498            # This is dCA_dt
499            rate[i] = - U[i] * CA_grad[i] - Da * ((1 - CA[i]) \
500                                    * coA[i] + lambda_ * CA[i] * coC[i])
501
502            # This is dCC_dt
503            rate[no_depths + i] = - U[i] * CC_grad[i] + Da * (lambda_ * \
504                                    (1 - CC[i]) * coC[i] + CC[i] * coA[i])              
505
506            # This is dcCa_dt
507            rate[2 * no_depths + i] =  helper_cCa_grad[i]/Phi[i] - W[i] * \
508                                        cCa_grad[i] + Da * one_minus_Phi[i] * \
509                                        (delta - cCa[i]) * common_helper3[i] \
510                                        /Phi[i]                                 
511
512            # This is dcCO3_dt
513            rate[3 * no_depths + i] =  helper_cCO3_grad[i]/Phi[i] - W[i] * \
514                                        cCO3_grad[i] + Da * one_minus_Phi[i] * \
515                                        (delta - cCO3[i]) * common_helper3[i] \
516                                        /Phi[i]                       
517
518            # This is dPhi_dt
519            rate[4 * no_depths + i] = - (dW_dx[i] * Phi[i] + W[i] * Phi_grad[i]) \
520                                      + dPhi[i] * Phi_laplace[i] + Da * one_minus_Phi[i] \
521                                      * common_helper3[i] 
522
523        return rate

199        # First, extract the porosity at the bottom of the system.

209    def calculate_sigma(Peclet, W_data, Peclet_min, Peclet_max):
210        ''' Calculate sigma following formula 8.73 from Boudreau:
211        "Diagenetic Models and their implementation"
212
213        Parameters
214        ----------
215        Peclet: ndarray(dtype=float, ndim=1)
216            Array along depth of the Peclet numbers.
217        W-data: ndarray(dtype=float, ndim=1)
218            Array along depth of the velocity of the pore water 
219            (counted positive downwards)
220        Peclet_min: float
221            Lower limit for calculating 1/tanh(Peclet)
222        Peclet_max: float
223            Upper limit for calculating 1/tanh(Peclet)
224
225        Returns
226        -------
227        sigma: ndarray(dtype=float, ndim=1)
228            Array along depth with sigma values
229        '''
230        sigma = np.empty(Peclet.size)
231        for i in range(sigma.size):
232            if np.abs(Peclet[i]) < Peclet_min:
233                sigma[i] = 0
234            elif np.abs(Peclet[i]) > Peclet_max:
235                sigma[i] = np.sign(W_data[i])
236            else:
237                sigma[i] = np.cosh(Peclet[i])/np.sinh(Peclet[i]) - \
238                    1/Peclet[i]
239        return sigma

241    def evolution_rate(self, state, t=0):
242        CA, CC, cCa, cCO3, Phi = state   
243
244        two_factors = cCa * cCO3
245        two_factors_upp_lim = two_factors.to_scalar(lambda f: np.fmin(f,1))
246        two_factors_low_lim = two_factors.to_scalar(lambda f: np.fmax(f,1))
247
248        three_factors = two_factors * self.KRat
249        three_factors_upp_lim = three_factors.to_scalar(lambda f: np.fmin(f,1))
250        three_factors_low_lim = three_factors.to_scalar(lambda f: np.fmax(f,1))
251
252        coA = CA * (((1 - three_factors_upp_lim) ** self.m2) * \
253                    (self.not_too_deep * self.not_too_shallow) - self.nu1 * \
254                    (three_factors_low_lim - 1) ** self.m1)
255 
256        coC = CC * (((two_factors_low_lim - 1) ** self.n1) - self.nu2 * \
257                    (1 - two_factors_upp_lim) ** self.n2)
258
259        F = 1 - np.exp(10 - 10 / Phi)
260        one_minus_Phi = 1 - Phi
261        U = self.presum + self.rhorat * Phi ** 3 * F /one_minus_Phi
262
263        # Choose either forward or backward differencing for CA and CC
264        # depending on the sign of U
265
266        CA_grad_back = CA.apply_operator("grad_back", self.bc_CA)
267        CA_grad_forw = CA.apply_operator("grad_forw", self.bc_CA)
268        CA_grad = ScalarField(state.grid, np.where(U.data>0, CA_grad_back.data, \
269            CA_grad_forw.data))
270
271        CC_grad_back = CC.apply_operator("grad_back", self.bc_CC)
272        CC_grad_forw = CC.apply_operator("grad_forw", self.bc_CC)
273        CC_grad = ScalarField(state.grid, np.where(U.data>0, CC_grad_back.data, \
274            CC_grad_forw.data))
275
276        W = self.presum - self.rhorat * Phi ** 2 * F
277        
278        dCA_dt = - U * CA_grad - self.Da * ((1 - CA) \
279                 * coA + self.lambda_ * CA * coC)
280
281        dCC_dt = - U * CC_grad + self.Da * (self.lambda_ \
282                 * (1 - CC) * coC + CC * coA)
283
284        # Implementing equation 6 from l'Heureux.
285        denominator = 1 - 2 * np.log(Phi)
286
287        # Fiadeiro-Veronis scheme for equations 42 and 43
288        # from l'Heureux. 
289        common_Peclet  = W * self.delta_x / 2. 
290        Peclet_cCa =  common_Peclet * denominator/ self.dCa       
291        sigma_cCa_data = LMAHeureuxPorosityDiff.calculate_sigma(\
292            Peclet_cCa.data, W.data, self.Peclet_min, self.Peclet_max)
293        sigma_cCa = ScalarField(state.grid, sigma_cCa_data)
294
295        Peclet_cCO3 = common_Peclet * denominator/ self.dCO3      
296        sigma_cCO3_data = LMAHeureuxPorosityDiff.calculate_sigma(\
297            Peclet_cCO3.data, W.data, self.Peclet_min, self.Peclet_max)
298        sigma_cCO3 = ScalarField(state.grid, sigma_cCO3_data)
299
300        # dPhi = self.auxcon * F * (Phi ** 3) / one_minus_Phi
301        dPhi = self.dPhi_fixed
302
303        Peclet_Phi = common_Peclet / dPhi
304        sigma_Phi_data = LMAHeureuxPorosityDiff.calculate_sigma(\
305            Peclet_Phi.data, W.data, self.Peclet_min, self.Peclet_max)
306        sigma_Phi = ScalarField(state.grid, sigma_Phi_data)
307
308        cCa_grad_back = cCa.apply_operator("grad_back", self.bc_cCa)
309        cCa_grad_forw = cCa.apply_operator("grad_forw", self.bc_cCa)
310        cCa_grad = 0.5 * ((1-sigma_cCa) * cCa_grad_forw +\
311             (1+sigma_cCa) * cCa_grad_back)
312
313        cCO3_grad_back = cCO3.apply_operator("grad_back", self.bc_cCO3)
314        cCO3_grad_forw = cCO3.apply_operator("grad_forw", self.bc_cCO3)
315        cCO3_grad = 0.5 * ((1-sigma_cCO3) * cCO3_grad_forw +\
316             (1+sigma_cCO3) * cCO3_grad_back)
317
318        Phi_grad_back = Phi.apply_operator("grad_back", self.bc_Phi)
319        Phi_grad_forw = Phi.apply_operator("grad_forw", self.bc_Phi)
320        Phi_grad = 0.5 * ((1-sigma_Phi) * Phi_grad_forw +\
321             (1+sigma_Phi) * Phi_grad_back)
322
323        Phi_denom = Phi/denominator
324        grad_Phi_denom = Phi_grad * (denominator + 2) / denominator ** 2
325
326        common_helper = coA - self.lambda_ * coC
327
328        dcCa_dt = (cCa_grad * grad_Phi_denom + Phi_denom * cCa.laplace(self.bc_cCa)) \
329                  * self.dCa /Phi -W * cCa_grad \
330                  + self.Da * one_minus_Phi * (self.delta - cCa) * common_helper / Phi
331
332        dcCO3_dt = (cCO3_grad * grad_Phi_denom + Phi_denom * cCO3.laplace(self.bc_cCO3)) \
333                   * self.dCO3/Phi -W * cCO3_grad \
334                   + self.Da * one_minus_Phi * (self.delta - cCO3) * common_helper / Phi
335
336        dW_dx = -self.rhorat * Phi_grad * (2 * Phi * F + 10 * (F - 1))
337
338        # This is closer to the original form of (43) from l' Heureux than
339        # the Matlab implementation.
340        dPhi_dt = - (Phi * dW_dx + W * Phi_grad) \
341                  + dPhi * Phi.laplace(self.bc_Phi) \
342                  + self.Da * one_minus_Phi * common_helper
343
344        return FieldCollection([dCA_dt, dCC_dt, dcCa_dt, dcCO3_dt, dPhi_dt])

347        """ the numba-accelerated evolution equation """

360        lambda_ = self.lambda_
361        Da = self.Da
362        dCa = self.dCa
363        dCO3 = self.dCO3
364        delta = self.delta
365        # The following three numbers are also needed for Fiadeiro-Veronis.
366        dPhi_fixed = self.dPhi_fixed
367        Peclet_min = self.Peclet_min
368        Peclet_max = self.Peclet_max
369        delta_x = state.grid._axes_coords[0][1] - state.grid._axes_coords[0][0]
370        grad_back_CA = state.grid.make_operator("grad_back", bc = self.bc_CA)
371        grad_forw_CA = state.grid.make_operator("grad_forw", bc = self.bc_CA)
372        grad_back_CC = state.grid.make_operator("grad_back", bc = self.bc_CC)
373        grad_forw_CC = state.grid.make_operator("grad_forw", bc = self.bc_CC)
374        grad_back_cCa = state.grid.make_operator("grad_back", bc = self.bc_cCa)
375        grad_forw_cCa = state.grid.make_operator("grad_forw", bc = self.bc_cCa)
376        laplace_cCa = state.grid.make_operator("laplace", bc = self.bc_cCa)
377        grad_back_cCO3 = state.grid.make_operator("grad_back", bc = self.bc_cCO3)
378        grad_forw_cCO3 = state.grid.make_operator("grad_forw", bc = self.bc_cCO3)
379        laplace_cCO3 = state.grid.make_operator("laplace", bc = self.bc_cCO3)
380        grad_back_Phi = state.grid.make_operator("grad_back", bc = self.bc_Phi)
381        grad_forw_Phi = state.grid.make_operator("grad_forw", bc = self.bc_Phi)
382        laplace_Phi = state.grid.make_operator("laplace", bc = self.bc_Phi)
383
384        @njit
385        def pde_rhs(state_data, t=0):
386            """ compiled helper function evaluating right hand side """
387            # Instead of the default central differenced gradient from py-pde
388            # construct forward and backward differenced gradients and apply
389            # either one of them, based on the sign of U.
390            CA = state_data[0]
391            CA_grad_back = grad_back_CA(CA)
392            CA_grad_forw = grad_forw_CA(CA)
393
394            CC = state_data[1]
395            CC_grad_back = grad_back_CC(CC)
396            CC_grad_forw = grad_forw_CC(CC)
397
398            cCa = state_data[2]
399            cCa_grad_back = grad_back_cCa(cCa)
400            cCa_grad_forw = grad_forw_cCa(cCa)
401            cCa_laplace = laplace_cCa(cCa)
402
403            cCO3 = state_data[3]
404            cCO3_grad_back = grad_back_cCO3(cCO3)
405            cCO3_grad_forw = grad_forw_cCO3(cCO3)
406            cCO3_laplace = laplace_cCO3(cCO3)
407
408            Phi = state_data[4]
409            Phi_grad_back = grad_back_Phi(Phi)
410            Phi_grad_forw = grad_forw_Phi(Phi)
411            Phi_laplace = laplace_Phi(Phi)
412
413            rate = np.empty_like(state_data)
414
415            # state_data.size should be the same as len(CA) or len(CC), check this.
416            # So the number of depths, really.
417            no_depths = state_data[0].size
418
419            denominator = np.empty(no_depths)
420            common_helper1 = np.empty(no_depths)
421            common_helper2 = np.empty(no_depths)
422            helper_cCa_grad = np.empty(no_depths)
423            helper_cCO3_grad = np.empty(no_depths)
424            F = np.empty(no_depths)
425            U = np.empty(no_depths)
426            W = np.empty(no_depths)
427            two_factors = np.empty(no_depths)
428            two_factors_upp_lim = np.empty(no_depths)
429            two_factors_low_lim = np.empty(no_depths)
430            three_factors = np.empty(no_depths)
431            three_factors_upp_lim = np.empty(no_depths)
432            three_factors_low_lim = np.empty(no_depths)
433            coA = np.empty(no_depths)
434            coC = np.empty(no_depths)
435            common_helper3 = np.empty(no_depths)
436            dPhi = np.empty(no_depths)
437            dW_dx = np.empty(no_depths)
438            one_minus_Phi = np.empty(no_depths)
439            CA_grad = np.empty(no_depths)
440            CC_grad = np.empty(no_depths)
441            cCa_grad = np.empty(no_depths)
442            cCO3_grad = np.empty(no_depths)
443            Phi_grad = np.empty(no_depths)            
444
445            for i in range(no_depths):
446                F[i] = 1 - np.exp(10 - 10 / Phi[i])
447
448                U[i] = presum + rhorat * Phi[i] ** 3 * F[i]/ (1 - Phi[i])
449
450                if U[i] > 0:
451                    CA_grad[i] = CA_grad_back[i]
452                    CC_grad[i] = CC_grad_back[i]
453                else:
454                    CA_grad[i] = CA_grad_forw[i]
455                    CC_grad[i] = CC_grad_forw[i]
456
457                W[i] = presum - rhorat * Phi[i] ** 2 * F[i]
458
459                # Implementing equation 6 from l'Heureux.
460                denominator[i] = 1 - 2 * np.log(Phi[i])
461
462                # Fiadeiro-Veronis scheme for equations 42 and 43
463                # from l'Heureux. 
464                Peclet_cCa = W[i] * delta_x * denominator[i]/ (2. * dCa )
465                if np.abs(Peclet_cCa) < Peclet_min:
466                    sigma_cCa = 0
467                elif np.abs(Peclet_cCa) > Peclet_max:
468                    sigma_cCa = np.sign(W[i])
469                else:
470                     sigma_cCa = np.cosh(Peclet_cCa)/np.sinh(Peclet_cCa) - \
471                        1/Peclet_cCa
472
473                Peclet_cCO3 = W[i] * delta_x * denominator[i]/ (2. * dCO3)
474                if np.abs(Peclet_cCO3) < Peclet_min:
475                    sigma_cCO3 = 0

512                three_factors_upp_lim[i] = min(three_factors[i],1)

520                    (1 - two_factors_upp_lim[i]) ** n2)

524                dW_dx[i] = -rhorat * Phi_grad[i] * (2 * Phi[i] * F[i] + 10 * (F[i] - 1))

526                # This is dCA_dt
527                rate[0][i] = - U[i] * CA_grad[i] - Da * ((1 - CA[i]) \
528                             * coA[i] + lambda_ * CA[i] * coC[i])
529
530                # This is dCC_dt
531                rate[1][i] = - U[i] * CC_grad[i] + Da * (lambda_ * \
532                             (1 - CC[i]) * coC[i] + CC[i] * coA[i])              
533
534                # This is dcCa_dt
535                rate[2][i] =  helper_cCa_grad[i]/Phi[i] - W[i] * \
536                              cCa_grad[i] + Da * one_minus_Phi[i] * \
537                              (delta - cCa[i]) * common_helper3[i] \
538                              /Phi[i]                                 
539
540                # This is dcCO3_dt
541                rate[3][i] =  helper_cCO3_grad[i]/Phi[i] - W[i] * \
542                              cCO3_grad[i] + Da * one_minus_Phi[i] * \
543                              (delta - cCO3[i]) * common_helper3[i] \
544                              /Phi[i]                       
545
546                # This is dPhi_dt
547                rate[4][i] = - (dW_dx[i] * Phi[i] + W[i] * Phi_grad[i]) \
548                             + dPhi[i] * Phi_laplace[i] + Da * one_minus_Phi[i] \
549                             * common_helper3[i]
